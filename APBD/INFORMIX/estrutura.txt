    funcodigo INT NOT NULL DEFAULT 0,
    funnome VARCHAR(50) NOT NULL,
    funsalario DECIMAL(6, 2) NOT NULL,
    funbaicodigo INT NOT NULL,
    funcodgerente INT DEFAULT NULL,
    fundtdem DATE DEFAULT NULL,

    FOREIGN KEY (funcodgerente) REFERENCES funcionario (funcodigo),
    FOREIGN KEY (funestcodigo) REFERENCES estadocivil (estcodigo)
);
CREATE INDEX idx_funbaicodigo ON funcionario (funbaicodigo);
CREATE INDEX idx_funcodgerente ON funcionario (funcodgerente);
CREATE INDEX idx_funestcodigo ON funcionario (funestcodigo);
